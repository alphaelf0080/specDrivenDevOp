🔄 詳細整合流程

📤 資料匯出流程
1
🎨
PSD 檔案分析
使用 Asset Exporter 掃描 Photoshop 檔案，識別所有 UI 元件的圖層結構
2
📊
資料結構化
將圖層資訊轉換為 JSON 格式，包含座標、尺寸、層級關係等遊戲開發所需資訊
3
💾
assetData.js 生成
輸出標準化的 JavaScript 資料檔案，可直接被 Cocos Creator 專案引用
4
🖼️
圖片素材匯出
同時批次匯出所有圖層為 PNG/JPG 圖片檔案，與資料檔案保持命名一致性
📥 Cocos Creator 整合
1
📁
資源匯入
將 assetData.js 和圖片素材匯入到 Cocos Creator 專案的 assets 資料夾
2
🔧
資料解析腳本
建立 TypeScript/JavaScript 腳本讀取 assetData，解析圖層資訊和位置關係
3
🎮
場景自動建構
根據 assetData 中的座標和層級資訊，自動建立遊戲場景中的 UI 元件
4
✨
互動邏輯綁定
為自動建立的 UI 元件添加點擊、動畫等遊戲邏輯和互動功能

💻 程式碼整合範例

📋 assetData.js 結構範例

const assetData = { "projectName": "GameUI_Design", "layers": [ { "id": 1001, "name": "main_background", "kind": "pixel", "visible": true, "opacity": 100, "bounds": { "left": 0, "top": 0, "right": 1920, "bottom": 1080 }, "position": { "x": 960, "y": 540 }, "originWidth": 1920, "originHeight": 1080, "scaledWidth": 1920, "scaledHeight": 1080, "parent": null, "depth": 0, "blendMode": "normal" }, { "id": 1002, "name": "start_button", "kind": "smartObject", "visible": true, "opacity": 100, "bounds": { "left": 810, "top": 640, "right": 1110, "bottom": 740 }, "position": { "x": 960, "y": 690 }, "originWidth": 300, "originHeight": 100, "scaledWidth": 300, "scaledHeight": 100, "parent": "main_background", "depth": 1, "blendMode": "normal" } ] }; export default assetData;

🔧 Cocos Creator 整合腳本範例

import { _decorator, Component, Node, Sprite, UITransform, Vec3 } from 'cc'; import assetData from './assetData'; const { ccclass, property } = _decorator; @ccclass('UIBuilder') export class UIBuilder extends Component { @property(Node) uiRoot: Node = null!; start() { this.buildUIFromAssetData(); } private buildUIFromAssetData() { console.log('🚀 開始建構 UI，共有', assetData.layers.length, '個圖層'); assetData.layers.forEach(layerData => { this.createUIElement(layerData); }); } private createUIElement(layerData: any) { // 建立節點 const node = new Node(layerData.name); // 設定位置 (Photoshop座標轉換為Cocos座標) const x = layerData.position.x - 960; // 轉換為中心座標 const y = 540 - layerData.position.y; // Y軸翻轉 node.setPosition(new Vec3(x, y, 0)); // 設定尺寸 const uiTransform = node.addComponent(UITransform); uiTransform.setContentSize(layerData.scaledWidth, layerData.scaledHeight); // 設定透明度 node.layer = layerData.opacity / 100; // 載入對應的圖片資源 const sprite = node.addComponent(Sprite); this.loadSpriteFrame(sprite, layerData.name); // 加入到場景 this.uiRoot.addChild(node); console.log(`✅ 已建立 UI 元件: ${layerData.name} at (${x}, ${y})`); } private loadSpriteFrame(sprite: Sprite, imageName: string) { // 動態載入圖片資源 const imagePath = `images/${imageName}/spriteFrame`; resources.load(imagePath, SpriteFrame, (err, spriteFrame) => { if (err) { console.error(`❌ 載入圖片失敗: ${imageName}`, err); return; } sprite.spriteFrame = spriteFrame; console.log(`🖼️ 圖片載入成功: ${imageName}`); }); } }
